1.二分查找

public int binarySearch(int[] nums, int target){
  int low = 0, high = nums.length - 1;
  while(low < high){
    int mid = (high - low) / 2 + low;
    if(nums[mid] < target){
      low = mid + 1; 
    }else{
      high = mid;
    }
  }
  return low;
}
// 大于的最后一个数（<=，不小于的最后一个数）
public int binarySearch(int[] nums, int target){
  int low = 0, high = nums.length - 1;
  while(low < high){
    int mid = (low + high + 1) >> 1;
    if(nums[mid] < target){
      low = mid;
    }else{
      high = mid - 1;
    }
  }
  return high;
}

2.并查集

// 并查集模版
public class UnionFind {
  // 定义保存节点的数组
  int[] parent;
  int[] rank; // 当前节点的秩
  int n;
  // 当前联通分量的数量
  int setCount;

  // 构造函数
  public UnionFind(int n) {
    this.n = n;
    this.setCount = n;
    this.parent = new int[n];
    this.rank = new int[n];
    // 初始化
    Arrays.fill(rank, 1);
    // 初始化头节点(i 和 -1 两种写法)
    for(int i=0; i<n; i++) parent[i] = i;
  }

  // 查找操作
  public int find(int index) {
       // 路经压缩
       if (parent[index] != index) {
         parent[index] = this.find(parent[index]);
       }
       return parent[index];
  }

  // 合并操作：是对边的信息进行操作，不是对点
  public boolean unionSet(int x, int y) {
    int x_root = this.find(x);
    int y_root = this.find(y);
    if (x_root == y_root) {
      return false;
    }
    //  按秩合并：将秩小的集合合并到秩大的集合中
    if (rank[x_root] > rank[y_root]) {
      parent[y_root] = x_root;
    } else if (rank[x_root] < rank[y_root]) {
      parent[x_root] = y_root;
    } else {
      rank[x_root]++;
      parent[y_root] = x_root;
    }
    this.setCount--;
    return true;
  }

  // 获取联通分量的数量
  public int getSetCount() {
    return this.setCount;
  }
}

5.Trie：前缀树

在计算机科学中，trie，又称前缀树或字典树，是一种有序树，用于保存关联数组，其中的键通常是字符串。与二叉查找树不同，键不是直接保存在节点中，而是由节点在树中的位置决定。一个节点的所有子孙都有相同的前缀，也就是这个节点对应的字符串，而根节点对应空字符串。一般情况下，不是所有的节点都有对应的值，只有叶子节点和部分内部节点所对应的键才有相关的值。 {inn, int, tea, ten, to}

class Trie{
  // 表示当前字母后面的字母
  private Trie[] children;
  private boolean isEnd;
  
  public Trie(){
    children = new Trie[26];
    isEnd = false;
  }
  
  public void insert(String word){
    Trie node = this; // 根结点
    for(int i=0; i<word.length(); i++){
      char ch = word.charAt(i);
      int index = ch - 'a';
      if(node.children[index] == null){
        node.children[index] = new Trie();
      }
      node = node.children[index];
    }
    // 新插入时最后一个字母节点的isEnd为true
    node.isEnd = true;
  }
  
  public boolean search(String word){
    Trie node = searchPrefix(word);
    return node != null && node.isEnd;
  }
  
  public boolean startsWith(String prefix){
    return searchPrefix(prefix) != null;
  }
  
  private Trie searchPrefix(String prefix){
    Trie node = this;
    for(int i=0 ;i<prefix.length(); i++){
      char ch = prefix.charAt(i);
      int index = ch - 'a';
      // 无当前字符
      if(node.children[index] == null){
        return null;
      }
      node = node.children[index];
    }
    return node;
  }
}

9.求逆元法｜快速幂

mod为质数，则i 的逆元就是$\frac{1}{i}=i^{mod-2}\% mod$

用快速幂求$i^{mod-2}$，算法时间复杂度为$O(logN)$

static final int MOD = 998244353;
static final int N = (int) 1e5 + 6;
static long[] fact = new long[N];
static long[] inFact = new long[N]; // 存储 i 的逆元
static long quick_pow(long a, long b){
  long ret = 1;
  while (b > 0){
    if((b & 1) == 1) ret = ret * a % MOD;
    a = a * a % MOD;
    b >>= 1; // b /= 2;
  }
  return ret % MOD;
}
inv(i) = quick_pow(i, MOD-2) % MOD;

10.组合数模板

$C_{a}^{b} = \frac{a!}{(a - b)!\ b!}$

$O(NlogN)$

static final int MOD = 998244353;
static final int N = (int) 1e5 + 6;
static long[] fact = new long[N]; //  存储阶乘
static long[] inFact = new long[N]; // 存储阶乘逆元
static long quick_pow(long a, long b){
  long ret = 1;
  while (b > 0){
    if((b & 1) == 1) ret = ret * a % MOD;
    a = a * a % MOD;
    b >>= 1; // b /= 2;
  }
  return ret % MOD;
}

static long comb(int a, int b){
  // 初始化
	fact[0] = inFact[0] = 1;
  for(int i=1; i<N; i++){
    fact[i] = fact[i-1] * i % MOD;
    inFact[i] = inFact[i-1] * quick_pow(i, MOD-2) % MOD;
  }
  return fact[a] * inFact[a - b] % MOD * inFact[b] % MOD;
}

11.树状数组

int N = (int)1e5 + 6; // 要看数据范围
int[] a = new int[N];
int[] c = new int[N];
int[] prefix = new int[N]; // 要看数据范围切换为 long
int n;

public int lowbit(int x){return x & -x;}

public void init(){
  for(int i=1; i<=n; i++){
    prefix[i] = prefix[i-1] + a[i-1];
    c[i] = prefix[i] - prefix[i - lowbit(i)];
  }
}

public void add(int x, int k){
  while(x <= n){
    c[x] += k;
    x += lowbit(x);
  }
}

public int getSum(int x){
  int ans = 0;
  while(x > 0){
    ans += c[x];
    x -= lowbit(x);
  }
  return ans;
}
